#include<stdio.h>
#include<malloc.h>
int main(){
	char * p;
	FILE * fp=fopen("c:\\test.txt","r");
	if(fp == NULL){
		printf("파일을 열수 없다\n");
		return -1;
	}
	fseek(fp,0,SEEK_END);
	int fsize = ftell(fp);
	fseek(fp,0,SEEK_SET);

	p = (char*)malloc(fsize);
	for(int i=0;i<fsize;i++){
		p[i]=0;
	}
	fread(p,1,fsize,fp);
	printf("%d\n",fsize);
	printf("%s",p);

	fclose(fp);
	free(p);

}
