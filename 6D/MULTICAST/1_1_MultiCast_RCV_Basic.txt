#pragma comment(lib, "ws2_32.lib")
#include <winsock2.h>
#include <stdio.h>
#include <ws2tcpip.h>

int main()
{
	WSADATA wsa;
	
	/* 윈속 dll 로드*/ 
	WSAStartup( MAKEWORD(2,2) ,&wsa);

	/* 소켓 생성*/
	SOCKET s;
	s = socket(AF_INET, SOCK_DGRAM,0);
	if(s == INVALID_SOCKET){
		printf("socket error!\n");
		return -1;
	}
	
	/*주소 결합*/
	//struct sockaddr_in
	SOCKADDR_IN SRVaddr;
	memset(&SRVaddr, 0, sizeof(SRVaddr));

	SRVaddr.sin_family = AF_INET;
	SRVaddr.sin_port = htons(12345);
	SRVaddr.sin_addr.s_addr = inet_addr("192.168.91.132");
	//SRVaddr.sin_addr.S_un.S_addr = 아이피주소;

	int errch = 0;
	errch = bind(s, (SOCKADDR*)&SRVaddr, sizeof(SRVaddr));
	if(errch == SOCKET_ERROR){
		printf("bind error!\n");
		return -1;
	}
	
	/*멀티캐스트 가입*/
	struct ip_mreq MR;
	MR.imr_interface.s_addr = inet_addr("192.168.91.132");
	MR.imr_multiaddr.s_addr = inet_addr("224.2.2.2");

	setsockopt(s,IPPROTO_IP, IP_ADD_MEMBERSHIP,(char*)&MR,sizeof(MR));


	/*데이터 수신*/
	char recvbuf[1024];
	memset(recvbuf, 0, sizeof(recvbuf));
	
	SOCKADDR_IN CLTaddr;
	int CLTlen = sizeof(CLTaddr);
	memset(&CLTaddr, 0, CLTlen);

	int recvlen=0;
recvlen = recvfrom(s,recvbuf,sizeof(recvbuf),0,(SOCKADDR*)&CLTaddr, &CLTlen);
	printf("수신 데이터 크기 : %d\n", recvlen);
	printf("수신 데이터  : %s\n", recvbuf);

	/*멀티캐스트 탈퇴*/
	setsockopt(s,IPPROTO_IP, IP_DROP_MEMBERSHIP, (char*)&MR, sizeof(MR));

	/*소켓 종료 및 dll 언로드*/
	closesocket(s);
	WSACleanup();
	return 0;
}
