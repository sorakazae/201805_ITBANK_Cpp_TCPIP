#pragma comment(lib, "ws2_32.lib")
#include <winsock2.h>
#include <stdio.h>
struct test{
	int num_1;
	char op;
	int num_2;
};
int main()
{
	WSADATA wsa;
	WSAStartup(MAKEWORD(2,2),&wsa);
	SOCKET s;
	s = socket(AF_INET,SOCK_DGRAM,0);
	if( s == INVALID_SOCKET){
		printf("sokcet_error!!!!");
		return -1;
	}
	SOCKADDR_IN addr;
	addr.sin_family = AF_INET;
	addr.sin_port = htons(12345);
	addr.sin_addr.s_addr = inet_addr("192.168.10.128");
	
	struct test op_;
	printf("°ª ÀÔ·Â : ");
	scanf("%d%c%d",&op_.num_1,&op_.op,&op_.num_2);

	sendto(s,(char*)&op_,sizeof(op_),0,(sockaddr*)&addr,sizeof(addr));
	char str[1024]={0};
	recvfrom(s,str,sizeof(str),0,0,0);
	printf("%d %c %d = %s\n",op_.num_1,op_.op,op_.num_2,str);
	closesocket(s);
	WSACleanup();
	return 0;
}
